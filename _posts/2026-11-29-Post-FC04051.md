``` python
import turtle as t
import random
import math

t.tracer(0)
t.speed(0)
sc=t.Screen()
sc.setup(1600,800)
sc.bgcolor('lightblue')
t.hideturtle()
t.penup()
t.setpos(-20,-60)
t.begin_fill()
t.fillcolor('yellow')
t.circle(100)
t.end_fill()


def mos(x,y,l):
    t.setpos(x,y)
    t.setheading(0)
    t.begin_fill()
    t.fillcolor('peru')
    for i in range(3):
        t.fd(l)
        t.left(120)
    t.end_fill()

def mos2(x,y,l):
    t.setpos(x,y)
    a=math.acos(0.5)
    b=math.pi-a
    B=math.degrees(b)
    A=math.degrees(a)
    t.begin_fill()
    t.fillcolor('peru')
    t.fd(l/2)
    t.left(B)
    t.fd(l)
    t.left(2*A)
    t.fd(l)
    t.left(B)
    t.end_fill()

mos(-300,-100,300)
mos(-50,-100,300)
mos2(200,-100,400)
mos2(-400,-100,400)
mos(300,-100,300)
mos2(600,-100,400)
mos2(-550,-100,450)

def q():
    t.setpos(-800,-100)
    t.begin_fill()
    t.fillcolor("lightgreen")
    t.setheading(0)
    t.fd(1600)
    t.right(90)
    t.fd(300)
    t.right(90)
    t.fd(1600)
    t.right(90)
    t.fd(300)
    t.right(90)
    t.end_fill()
q()

def abr(x, y, size=40):
    t.color("white")


    t.penup()
    t.goto(x, y)
    t.begin_fill()
    t.circle(size)
    t.end_fill()

    t.penup()
    t.goto(x + size*0.8, y + size*0.4)
    t.begin_fill()
    t.circle(size*0.9)
    t.end_fill()

    
    t.penup()
    t.goto(x - size*0.8, y + size*0.4)
    t.begin_fill()
    t.circle(size*0.9)
    t.end_fill()

    
    t.penup()
    t.goto(x, y + size*0.7)
    t.begin_fill()
    t.circle(size*0.8)
    t.end_fill()
abr(-300, 250, 40)
abr(0, 200, 50)
abr(300, 260, 35)


def tree(l,s):
    t.color("green")
    if l>5:
        a=random.randint(15,20)
        a1=random.randint(15,20)
        x=random.uniform(0.5,0.9)
        t.pensize(max(1,s))
        t.pendown()
        t.forward(l)
        t.left(a)
        tree(x*l,s-1)
        t.right(a1+a)
        tree(x*l,s-1)
        t.left(a1)
        t.penup()
        t.backward(l)
    else:
        t.pendown()
        t.begin_fill()
        t.color("red")
        t.circle(3)       
        t.end_fill()
        t.penup()
t.setheading(90) 
for i in range(-500,501,250):
    if i==0:
        continue
    t.setpos(i,-350)
    tree(120,4)    
t.setpos(0,-350) 
tree(100,6)  


t.update()
t.mainloop()
``` 
